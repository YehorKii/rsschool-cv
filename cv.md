## Code Example

```ts
function sortArray(arr) {
    const filteredOdd = arr.filter(el => el % 2).sort((a, b) => a - b);
    let result = [...arr];
    for (let i = 0, x = 0; i < result.length; i++) {
        if (result[i] % 2) {
            result[i] = filteredOdd[x];
            x++;
        }
    }
    return result;
}
```
